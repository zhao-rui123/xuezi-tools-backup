// 时段类型定义
export type PeriodType = 'peak' | 'valley' | 'flat' | 'sharp' | 'deepValley';

// 时段配置接口

  start: number;  // 开始小时 (0-24)
  end: number;    // 结束小时 (0-24)
}

export interface PeriodConfig {
  type: PeriodType;
  ranges: TimeRange[];
}

export interface MonthConfig {
  months: number[];  // 适用月份 (1-12)
  periods: PeriodConfig[];
}

export interface ProvinceConfig {
  name: string;
  code: string;
  monthConfigs: MonthConfig[];
}

// ==================== 工具函数 ====================

/**
 * 根据省份代码获取配置
 */
export function getProvinceConfig(code: string): ProvinceConfig | undefined {
  return allProvinceConfigs.find(p => p.code === code);
}

/**
 * 根据月份获取对应的时段配置
 */
export function getMonthConfig(province: ProvinceConfig, month: number): MonthConfig | undefined {
  return province.monthConfigs.find(mc => mc.months.includes(month));
}

/**
 * 获取指定小时对应的时段类型
 */
export function getPeriodTypeAtHour(monthConfig: MonthConfig, hour: number): PeriodType {
  for (const period of monthConfig.periods) {
    for (const range of period.ranges) {
      if (hour >= range.start && hour < range.end) {
        return period.type;
      }
    }
  }
  return 'flat';  // 默认平时段
}

/**
 * 获取时段类型的中文名称
 */
export function getPeriodTypeName(type: PeriodType): string {
  const names: Record<PeriodType, string> = {
    peak: '高峰',
    valley: '低谷',
    flat: '平段',
    sharp: '尖峰',
    deepValley: '深谷'
  };
  return names[type] || '未知';
}
