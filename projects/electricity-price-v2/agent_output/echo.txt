// ==================== 工具函数 ====================

/**
 * 获取指定月份和小时的电价时段类型
 * @param policy 省份电价政策
 * @param month 月份 (1-12)
 * @param hour 小时 (0-23)
 * @returns 时段类型: 'sharp' | 'peak' | 'flat' | 'valley' | 'deepValley' | null
 */
export function getPricePeriodType(
  policy: ProvincePricePolicy,
  month: number,
  hour: number
): 'sharp' | 'peak' | 'flat' | 'valley' | 'deepValley' | null {
  if (!policy.hasTimeOfUse) return null;

  const season = policy.seasons.find(s => s.months.includes(month));
  if (!season) return null;

  // 检查优先级: 尖峰 > 深谷 > 高峰 > 低谷 > 平段
  if (season.sharp && isHourInPeriods(hour, season.sharp)) return 'sharp';
  if (season.deepValley && isHourInPeriods(hour, season.deepValley)) return 'deepValley';
  if (season.peak && isHourInPeriods(hour, season.peak)) return 'peak';
  if (season.valley && isHourInPeriods(hour, season.valley)) return 'valley';
  if (season.flat && isHourInPeriods(hour, season.flat)) return 'flat';

  return null;
}

/**
 * 检查小时是否在时段列表中
 */
function isHourInPeriods(hour: number, periods: TimePeriod[]): boolean {
  return periods.some(p => hour >= p.start && hour < p.end);
}

/**
 * 根据省份代码获取电价政策
 */
export function getPolicyByCode(code: string): ProvincePricePolicy | undefined {
  return allPricePolicies.find(p => p.code === code);
}

/**
 * 根据省份名称获取电价政策
 */
export function getPolicyByName(name: string): ProvincePricePolicy | undefined {
  return allPricePolicies.find(p => p.name === name);
}
